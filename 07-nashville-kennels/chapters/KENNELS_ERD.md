Optional: Use the below to recreate the ERD at dbdiagram.io.
```
table animals {
  id int [PK]
  name varchar
  breed varchar
  customerId [FK]
  locationId [FK]
}

table locations {
  id int [PK]
  name varchar 
  address varchar 
}

table customers {
  id int [PK]
  name varchar 
  address varchar
}

table employees {
  id int [PK]
  name varchar 
  locationId [FK]
}


Ref: "employees"."locationId" > "locations"."id"

Ref: "locations"."id" < "animals"."locationId"

Ref: "customers"."id" < "animals"."customerId"
```
